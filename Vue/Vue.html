4.数据视图 - 监听属性 属性中的值需要v-model="xxdata"

修改这个输入 
       
        var vm = new Vue({
            el: "#data4",
            data: {
                data: "这个是测试数据"
            },
            watch: {
                data: function (newValue, oldValue) {
                    console.log("只要data数据改变，就会触发这个函数：修改前的数据为" + oldValue);
                    console.log("只要data数据改变，就会触发这个函数：修改后的数据为" + newValue);
                }
            }
        });
        //这个是外界手动触发函数
        vm.$watch("data", function (newValue, oldValue) {
            console.log("新数据" + newValue);
            console.log("就数据" + oldValue);
        })
